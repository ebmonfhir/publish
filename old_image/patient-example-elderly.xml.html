Example of elderly patient (id = "elderly")

<?xml version="1.0" encoding="UTF-8"?>

<Patient xmlns="http://hl7.org/fhir">
      <!--    simple elderly patient example    -->
  <id value="elderly"/> 
      <!--    Details of patient    -->
  <text> <status value="generated"/> <div xmlns="http://www.w3.org/1999/xhtml"><p> <b> Generated Narrative with Details</b> </p> <p> <b> id</b> : elderly</p> <p> <b> gender</b> : male</p> <p> <b> birthDate</b> : Mar 25, 1940</p> <h3> Contacts</h3> <table> <tr> <td> -</td> <td> <b> Relationship</b> </td> <td> <b> Name</b> </td> <td> <b> Telecom</b> </td> </tr> <tr> <td> *</td> <td> Daughter <span> (Details : {SNOMED CT code '160454000' = 'FH: Daughter', given as 'Daughter'})</span> </td> <td> Jane Doe (MAIDEN)</td> <td> ph: +5555555555(MOBILE)</td> </tr> </table> </div> </text> <gender value="male"/> 
  <birthDate value="1940-03-25">
    <extension url="http://hl7.org/fhir/StructureDefinition/patient-birthTime">
      <valueDateTime value="1940-03-25T17:11:00+01:00"/> 
    </extension> 
  </birthDate> 
      <!--    contact info of caretaker    -->
  <contact> 
    <relationship> 
      <coding> 
        <system value="http://snomed.info/sct"/> 
        <code value="160454000"/> 
        <display value="Daughter"/> 
      </coding> 
    </relationship> 
    <name> 
      <use value="maiden"/> 
      <family value="Doe"/> 
      <given value="Jane"/> 
    </name> 
    <telecom> 
      <system value="phone"/> 
      <value value="+5555555555"/> 
      <use value="mobile"/> 
    </telecom> 
  </contact> 
</Patient>
